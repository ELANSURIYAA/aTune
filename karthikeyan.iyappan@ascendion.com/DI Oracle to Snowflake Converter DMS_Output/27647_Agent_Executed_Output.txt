-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Stored Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE to Snowflake stored procedure using JavaScript
-- - Replaced Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) with Snowflake JavaScript equivalents
-- - Converted SYSTIMESTAMP to CURRENT_TIMESTAMP()
-- - Replaced USER with CURRENT_USER()
-- - Converted DBMS_OUTPUT.PUT_LINE to Snowflake result set returns and logging
-- - Translated Oracle procedure calls to Snowflake CALL statements
-- - Converted Oracle NVL function to COALESCE
-- - Replaced Oracle EXTRACT function with Snowflake DATEDIFF for time calculations
-- - Converted Oracle INSERT INTO statements to Snowflake INSERT syntax
-- - Translated Oracle COMMIT to Snowflake transaction control
-- - Replaced Oracle exception handling (EXCEPTION WHEN OTHERS) with JavaScript try-catch blocks
-- - Converted SQLERRM and SQLCODE to JavaScript error handling
-- - Replaced RAISE_APPLICATION_ERROR with JavaScript throw statement
-- Major Risks / Checks :
-- - Validate transaction behavior differences between Oracle and Snowflake
-- - Validate error handling and exception propagation
-- - Validate timestamp precision and date arithmetic
-- - Validate procedure call execution and parameter passing
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
EXECUTE AS CALLER
AS
$$
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage = '';
    var v_ErrorNumber = 0;
    var v_ErrorSeverity = 0;
    var v_ErrorState = 0;
    var v_ErrorLine = 0;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser = '';
    var v_BatchID = '';
    
    try {
        // Get current user
        var userStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var userResult = userStmt.execute();
        if (userResult.next()) {
            v_CurrentUser = userResult.getColumnValue(1);
        }
        
        // Get start timestamp and create batch ID
        var timestampStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var timestampResult = timestampStmt.execute();
        if (timestampResult.next()) {
            v_StartTime = timestampResult.getColumnValue(1);
        }
        
        var batchStmt = snowflake.createStatement({sqlText: "SELECT TO_CHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var batchResult = batchStmt.execute();
        if (batchResult.next()) {
            v_BatchID = batchResult.getColumnValue(1);
        }
        
        // Log start of overall process
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', '================================================================================')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'bronze Layer ETL Pipeline - Started at: " + v_BatchID + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Executed by: " + v_CurrentUser + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Batch ID: " + v_BatchID + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', '================================================================================')"});
        
        // Load Table 1: bz_New_Monthly_HC_Report
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_SchTask('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_report_392_all('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_DimDate('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_holidays('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        snowflake.execute({sqlText: "CALL bronze.usp_Load_bz_holidays_India('" + v_BatchID + "')"});
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Calculate summary statistics from audit log
        var successStmt = snowflake.createStatement({
            sqlText: "SELECT COUNT(*), COALESCE(SUM(records_inserted), 0) FROM bronze.bz_Audit_Log WHERE batch_id = '" + v_BatchID + "' AND status = 'SUCCESS'"
        });
        var successResult = successStmt.execute();
        if (successResult.next()) {
            v_TablesSucceeded = successResult.getColumnValue(1);
            v_TotalRowsInserted = successResult.getColumnValue(2);
        }
        
        var failedStmt = snowflake.createStatement({
            sqlText: "SELECT COUNT(*), COALESCE(SUM(records_failed), 0) FROM bronze.bz_Audit_Log WHERE batch_id = '" + v_BatchID + "' AND status = 'FAILED'"
        });
        var failedResult = failedStmt.execute();
        if (failedResult.next()) {
            v_TablesFailed = failedResult.getColumnValue(1);
            v_TotalRowsFailed = failedResult.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // Calculate execution time
        var endTimeStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var endTimeResult = endTimeStmt.execute();
        if (endTimeResult.next()) {
            v_EndTime = endTimeResult.getColumnValue(1);
        }
        
        var execTimeStmt = snowflake.createStatement({
            sqlText: "SELECT DATEDIFF(SECOND, '" + v_StartTime + "', '" + v_EndTime + "')"
        });
        var execTimeResult = execTimeStmt.execute();
        if (execTimeResult.next()) {
            v_ExecutionTime = execTimeResult.getColumnValue(1);
        }
        
        // Log completion
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', '================================================================================')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'bronze Layer ETL Pipeline - Completed Successfully')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', '================================================================================')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'End Time: " + v_EndTime + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Execution Time: " + v_ExecutionTime + " seconds')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Tables Processed: " + v_TablesProcessed + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Tables Succeeded: " + v_TablesSucceeded + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Tables Failed: " + v_TablesFailed + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Total Rows Inserted: " + v_TotalRowsInserted + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', 'Total Rows Failed: " + (v_TotalRowsFailed || 0) + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'INFO', '" + v_ProcedureName + "', '================================================================================')"});
        
        // Insert master audit record
        var insertAuditStmt = snowflake.createStatement({
            sqlText: "INSERT INTO bronze.bz_Audit_Log (source_table, target_table, load_timestamp, start_timestamp, end_timestamp, processed_by, processing_time, status, records_processed, records_inserted, records_failed, batch_id, load_type, created_date) VALUES ('source_layer.*', 'bronze.*', '" + v_StartTime + "', '" + v_StartTime + "', '" + v_EndTime + "', '" + v_CurrentUser + "', " + v_ExecutionTime + ", '" + v_OverallStatus + "', " + v_TotalRowsProcessed + ", " + v_TotalRowsInserted + ", " + (v_TotalRowsFailed || 0) + ", '" + v_BatchID + "', 'FULL_REFRESH_ALL_TABLES', CURRENT_TIMESTAMP())"
        });
        insertAuditStmt.execute();
        
        // CHECK REQUIRED:
        // Snowflake uses auto-commit by default for stored procedures.
        // Oracle explicit COMMIT behavior may differ from Snowflake transaction handling.
        // Validate transaction isolation and commit behavior.
        
        return 'SUCCESS: bronze Layer ETL Pipeline completed successfully. Batch ID: ' + v_BatchID;
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        // Get end time
        var endTimeStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var endTimeResult = endTimeStmt.execute();
        if (endTimeResult.next()) {
            v_EndTime = endTimeResult.getColumnValue(1);
        }
        
        // Calculate execution time
        var execTimeStmt = snowflake.createStatement({
            sqlText: "SELECT DATEDIFF(SECOND, '" + v_StartTime + "', '" + v_EndTime + "')"
        });
        var execTimeResult = execTimeStmt.execute();
        if (execTimeResult.next()) {
            v_ExecutionTime = execTimeResult.getColumnValue(1);
        }
        
        v_OverallStatus = 'FAILED';
        
        // Log error
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', '================================================================================')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', 'ERROR in bronze Layer ETL Pipeline')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', '================================================================================')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', 'Error Number: " + v_ErrorNumber + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', 'Error Severity: " + v_ErrorSeverity + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', 'Error State: " + v_ErrorState + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', 'Error Line: " + v_ErrorLine + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', 'Error Message: " + v_ErrorMessage.replace(/'/g, "''") + "')"});
        snowflake.execute({sqlText: "INSERT INTO bronze.bz_Process_Log (log_timestamp, log_level, procedure_name, message) VALUES (CURRENT_TIMESTAMP(), 'ERROR', '" + v_ProcedureName + "', '================================================================================')"});
        
        // Insert error audit record
        var insertErrorAuditStmt = snowflake.createStatement({
            sqlText: "INSERT INTO bronze.bz_Audit_Log (source_table, target_table, load_timestamp, start_timestamp, end_timestamp, processed_by, processing_time, status, error_message, batch_id, load_type, created_date) VALUES ('source_layer.*', 'bronze.*', '" + v_StartTime + "', '" + v_StartTime + "', '" + v_EndTime + "', '" + v_CurrentUser + "', " + v_ExecutionTime + ", '" + v_OverallStatus + "', 'Error " + v_ErrorNumber + ": " + v_ErrorMessage.replace(/'/g, "''") + "', '" + v_BatchID + "', 'FULL_REFRESH_ALL_TABLES', CURRENT_TIMESTAMP())"
        });
        insertErrorAuditStmt.execute();
        
        // CHECK REQUIRED:
        // Snowflake uses auto-commit by default for stored procedures.
        // Oracle explicit COMMIT behavior may differ from Snowflake transaction handling.
        // Validate transaction isolation and commit behavior.
        
        // Re-throw error
        throw 'FAILED: ' + v_ErrorMessage;
    }
$$;
