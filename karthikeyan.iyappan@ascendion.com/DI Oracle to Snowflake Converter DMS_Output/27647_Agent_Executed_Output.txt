-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE into Snowflake CREATE OR REPLACE PROCEDURE with JavaScript
-- - Replaced Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) with Snowflake JavaScript equivalents
-- - Converted SYSTIMESTAMP to CURRENT_TIMESTAMP()
-- - Replaced USER with CURRENT_USER()
-- - Converted Oracle TO_CHAR date formatting to Snowflake TO_CHAR with equivalent format strings
-- - Replaced DBMS_OUTPUT.PUT_LINE with JavaScript return statements and logging approach
-- - Converted Oracle procedure calls into Snowflake CALL statements within JavaScript
-- - Replaced NVL with COALESCE in SQL queries
-- - Converted Oracle EXTRACT function for timestamp arithmetic to Snowflake DATEDIFF
-- - Translated Oracle INSERT INTO statements into Snowflake INSERT INTO with JavaScript SQL execution
-- - Replaced COMMIT with Snowflake auto-commit behavior
-- - Converted Oracle EXCEPTION block into JavaScript try-catch error handling
-- - Replaced SQLERRM and SQLCODE with JavaScript error object properties
-- - Converted RAISE_APPLICATION_ERROR into JavaScript throw statement
-- Major Risks / Checks :
-- - Validate timestamp arithmetic conversion from EXTRACT to DATEDIFF
-- - Validate error handling behavior differences between Oracle and Snowflake
-- - Validate transaction management (Oracle COMMIT vs Snowflake auto-commit)
-- - Validate procedure call syntax and parameter passing
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
EXECUTE AS CALLER
AS
$$
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage = '';
    var v_ErrorNumber = 0;
    var v_ErrorSeverity = 0;
    var v_ErrorState = 0;
    var v_ErrorLine = 0;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser = '';
    var v_BatchID = '';
    
    try {
        // Get current user
        var userStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var userResult = userStmt.execute();
        if (userResult.next()) {
            v_CurrentUser = userResult.getColumnValue(1);
        }
        
        // Get start timestamp and create batch ID
        var timestampStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var timestampResult = timestampStmt.execute();
        if (timestampResult.next()) {
            v_StartTime = timestampResult.getColumnValue(1);
        }
        
        var batchIdStmt = snowflake.createStatement({sqlText: "SELECT TO_CHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var batchIdResult = batchIdStmt.execute();
        if (batchIdResult.next()) {
            v_BatchID = batchIdResult.getColumnValue(1);
        }
        
        // Log start of overall process
        var logMessage = '';
        logMessage += '================================================================================\n';
        logMessage += 'bronze Layer ETL Pipeline - Started at: ' + v_BatchID + '\n';
        logMessage += 'Executed by: ' + v_CurrentUser + '\n';
        logMessage += 'Batch ID: ' + v_BatchID + '\n';
        logMessage += '================================================================================\n';
        logMessage += '\n';
        
        // Load Table 1: bz_New_Monthly_HC_Report
        var callStmt1 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report(?)", binds: [v_BatchID]});
        callStmt1.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask (excludes TIMESTAMP column)
        var callStmt2 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_SchTask(?)", binds: [v_BatchID]});
        callStmt2.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        var callStmt3 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info(?)", binds: [v_BatchID]});
        callStmt3.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        var callStmt4 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New(?)", binds: [v_BatchID]});
        callStmt4.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        var callStmt5 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_report_392_all(?)", binds: [v_BatchID]});
        callStmt5.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        var callStmt6 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne(?)", binds: [v_BatchID]});
        callStmt6.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        var callStmt7 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise(?)", binds: [v_BatchID]});
        callStmt7.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        var callStmt8 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_DimDate(?)", binds: [v_BatchID]});
        callStmt8.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        var callStmt9 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico(?)", binds: [v_BatchID]});
        callStmt9.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        var callStmt10 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada(?)", binds: [v_BatchID]});
        callStmt10.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        var callStmt11 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays(?)", binds: [v_BatchID]});
        callStmt11.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        var callStmt12 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_India(?)", binds: [v_BatchID]});
        callStmt12.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Calculate summary statistics from audit log
        var summarySuccessQuery = `
            SELECT 
                COUNT(*),
                COALESCE(SUM(records_inserted), 0)
            FROM bronze.bz_Audit_Log
            WHERE batch_id = ?
                AND status = 'SUCCESS'
        `;
        var summarySuccessStmt = snowflake.createStatement({sqlText: summarySuccessQuery, binds: [v_BatchID]});
        var summarySuccessResult = summarySuccessStmt.execute();
        if (summarySuccessResult.next()) {
            v_TablesSucceeded = summarySuccessResult.getColumnValue(1);
            v_TotalRowsInserted = summarySuccessResult.getColumnValue(2);
        }
        
        var summaryFailedQuery = `
            SELECT 
                COUNT(*),
                COALESCE(SUM(records_failed), 0)
            FROM bronze.bz_Audit_Log
            WHERE batch_id = ?
                AND status = 'FAILED'
        `;
        var summaryFailedStmt = snowflake.createStatement({sqlText: summaryFailedQuery, binds: [v_BatchID]});
        var summaryFailedResult = summaryFailedStmt.execute();
        if (summaryFailedResult.next()) {
            v_TablesFailed = summaryFailedResult.getColumnValue(1);
            v_TotalRowsFailed = summaryFailedResult.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // Calculate execution time
        var endTimeStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var endTimeResult = endTimeStmt.execute();
        if (endTimeResult.next()) {
            v_EndTime = endTimeResult.getColumnValue(1);
        }
        
        // CHECK REQUIRED:
        // Oracle EXTRACT function for timestamp arithmetic converted to Snowflake DATEDIFF.
        // Validate that execution time calculation preserves exact semantics.
        var execTimeQuery = "SELECT DATEDIFF(SECOND, ?, ?)";
        var execTimeStmt = snowflake.createStatement({sqlText: execTimeQuery, binds: [v_StartTime, v_EndTime]});
        var execTimeResult = execTimeStmt.execute();
        if (execTimeResult.next()) {
            v_ExecutionTime = execTimeResult.getColumnValue(1);
        }
        
        // Log completion
        logMessage += '\n';
        logMessage += '================================================================================\n';
        logMessage += 'bronze Layer ETL Pipeline - Completed Successfully\n';
        logMessage += '================================================================================\n';
        logMessage += 'End Time: ' + v_EndTime + '\n';
        logMessage += 'Execution Time: ' + v_ExecutionTime + ' seconds\n';
        logMessage += 'Tables Processed: ' + v_TablesProcessed + '\n';
        logMessage += 'Tables Succeeded: ' + v_TablesSucceeded + '\n';
        logMessage += 'Tables Failed: ' + v_TablesFailed + '\n';
        logMessage += 'Total Rows Inserted: ' + v_TotalRowsInserted + '\n';
        logMessage += 'Total Rows Failed: ' + (v_TotalRowsFailed || 0) + '\n';
        logMessage += '================================================================================\n';
        
        // Insert master audit record
        var insertAuditQuery = `
            INSERT INTO bronze.bz_Audit_Log (
                source_table,
                target_table,
                load_timestamp,
                start_timestamp,
                end_timestamp,
                processed_by,
                processing_time,
                status,
                records_processed,
                records_inserted,
                records_failed,
                batch_id,
                load_type,
                created_date
            )
            VALUES (
                'source_layer.*',
                'bronze.*',
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                'FULL_REFRESH_ALL_TABLES',
                CURRENT_TIMESTAMP()
            )
        `;
        var insertAuditStmt = snowflake.createStatement({
            sqlText: insertAuditQuery,
            binds: [
                v_StartTime,
                v_StartTime,
                v_EndTime,
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                v_TotalRowsProcessed,
                v_TotalRowsInserted,
                v_TotalRowsFailed,
                v_BatchID
            ]
        });
        insertAuditStmt.execute();
        
        // CHECK REQUIRED:
        // Snowflake uses auto-commit by default for stored procedures.
        // Oracle explicit COMMIT converted to Snowflake implicit commit behavior.
        
        return logMessage + '\nStatus: ' + v_OverallStatus;
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        var endTimeStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var endTimeResult = endTimeStmt.execute();
        if (endTimeResult.next()) {
            v_EndTime = endTimeResult.getColumnValue(1);
        }
        
        var execTimeQuery = "SELECT DATEDIFF(SECOND, ?, ?)";
        var execTimeStmt = snowflake.createStatement({sqlText: execTimeQuery, binds: [v_StartTime, v_EndTime]});
        var execTimeResult = execTimeStmt.execute();
        if (execTimeResult.next()) {
            v_ExecutionTime = execTimeResult.getColumnValue(1);
        }
        
        v_OverallStatus = 'FAILED';
        
        // Log error
        var errorLog = '';
        errorLog += '\n';
        errorLog += '================================================================================\n';
        errorLog += 'ERROR in bronze Layer ETL Pipeline\n';
        errorLog += '================================================================================\n';
        errorLog += 'Error Number: ' + v_ErrorNumber + '\n';
        errorLog += 'Error Severity: ' + v_ErrorSeverity + '\n';
        errorLog += 'Error State: ' + v_ErrorState + '\n';
        errorLog += 'Error Line: ' + v_ErrorLine + '\n';
        errorLog += 'Error Message: ' + v_ErrorMessage + '\n';
        errorLog += '================================================================================\n';
        
        // Insert error audit record
        var insertErrorAuditQuery = `
            INSERT INTO bronze.bz_Audit_Log (
                source_table,
                target_table,
                load_timestamp,
                start_timestamp,
                end_timestamp,
                processed_by,
                processing_time,
                status,
                error_message,
                batch_id,
                load_type,
                created_date
            )
            VALUES (
                'source_layer.*',
                'bronze.*',
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                ?,
                'FULL_REFRESH_ALL_TABLES',
                CURRENT_TIMESTAMP()
            )
        `;
        var insertErrorAuditStmt = snowflake.createStatement({
            sqlText: insertErrorAuditQuery,
            binds: [
                v_StartTime,
                v_StartTime,
                v_EndTime,
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                'Error ' + v_ErrorNumber + ': ' + v_ErrorMessage,
                v_BatchID
            ]
        });
        insertErrorAuditStmt.execute();
        
        // CHECK REQUIRED:
        // Oracle RAISE_APPLICATION_ERROR converted to JavaScript throw.
        // Validate error propagation behavior in Snowflake stored procedures.
        throw errorLog + '\nError: ' + v_ErrorMessage;
    }
$$;
