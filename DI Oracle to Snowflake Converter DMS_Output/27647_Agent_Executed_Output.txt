Description: Stored procedure to orchestrate full refresh load of all bronze layer tables with comprehensive audit logging and error handling

-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Stored Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE to Snowflake stored procedure using JavaScript
-- - Translated Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) to Snowflake JavaScript variables
-- - Replaced Oracle SYSTIMESTAMP with Snowflake CURRENT_TIMESTAMP()
-- - Converted Oracle USER function to Snowflake CURRENT_USER()
-- - Replaced Oracle TO_CHAR date formatting with Snowflake TO_VARCHAR and TO_CHAR equivalents
-- - Converted Oracle DBMS_OUTPUT.PUT_LINE to Snowflake JavaScript return statements and logging
-- - Translated Oracle procedure calls to Snowflake CALL statements within JavaScript
-- - Converted Oracle NVL function to Snowflake COALESCE
-- - Replaced Oracle EXTRACT function for timestamp arithmetic with Snowflake DATEDIFF
-- - Converted Oracle INSERT INTO statements to Snowflake INSERT syntax
-- - Translated Oracle COMMIT to Snowflake transaction control
-- - Converted Oracle exception handling (EXCEPTION WHEN OTHERS) to JavaScript try-catch blocks
-- - Replaced Oracle SQLERRM and SQLCODE with Snowflake JavaScript error handling
-- - Converted Oracle RAISE_APPLICATION_ERROR to Snowflake JavaScript throw statement
-- Major Risks / Checks :
-- - Validate VARCHAR2 to VARCHAR conversion and length limits
-- - Validate SYSTIMESTAMP to CURRENT_TIMESTAMP() timezone behavior
-- - Validate date arithmetic and EXTRACT to DATEDIFF conversion
-- - Validate NVL to COALESCE NULL handling semantics
-- - Validate transaction control and COMMIT behavior
-- - Validate error handling and exception propagation
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
EXECUTE AS CALLER
AS
$$
    // CHECK REQUIRED: Oracle VARCHAR2 converted to Snowflake VARCHAR - validate length limits and padding behavior
    // CHECK REQUIRED: Oracle NUMBER converted to Snowflake NUMBER - validate precision and scale
    // CHECK REQUIRED: Oracle TIMESTAMP converted to Snowflake TIMESTAMP - validate timezone behavior
    
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage = '';
    var v_ErrorNumber = 0;
    var v_ErrorSeverity = 0;
    var v_ErrorState = 0;
    var v_ErrorLine = 0;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser = '';
    var v_BatchID = '';
    
    try {
        // CHECK REQUIRED: Oracle SYSTIMESTAMP converted to Snowflake CURRENT_TIMESTAMP() - validate timezone behavior
        var stmt_start = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var result_start = stmt_start.execute();
        result_start.next();
        v_StartTime = result_start.getColumnValue(1);
        
        // CHECK REQUIRED: Oracle USER function converted to Snowflake CURRENT_USER() - validate behavior
        var stmt_user = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var result_user = stmt_user.execute();
        result_user.next();
        v_CurrentUser = result_user.getColumnValue(1);
        
        // CHECK REQUIRED: Oracle TO_CHAR date formatting converted to Snowflake TO_VARCHAR - validate format string compatibility
        var stmt_batch = snowflake.createStatement({sqlText: "SELECT TO_VARCHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var result_batch = stmt_batch.execute();
        result_batch.next();
        v_BatchID = result_batch.getColumnValue(1);
        
        // Log start of overall process
        var log_msg = '';
        log_msg += '================================================================================\n';
        log_msg += 'bronze Layer ETL Pipeline - Started at: ' + v_BatchID + '\n';
        log_msg += 'Executed by: ' + v_CurrentUser + '\n';
        log_msg += 'Batch ID: ' + v_BatchID + '\n';
        log_msg += '================================================================================\n';
        log_msg += '\n';
        
        // Load Table 1: bz_New_Monthly_HC_Report
        var stmt1 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report(?)", binds: [v_BatchID]});
        stmt1.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask (excludes TIMESTAMP column)
        var stmt2 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_SchTask(?)", binds: [v_BatchID]});
        stmt2.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        var stmt3 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info(?)", binds: [v_BatchID]});
        stmt3.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        var stmt4 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New(?)", binds: [v_BatchID]});
        stmt4.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        var stmt5 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_report_392_all(?)", binds: [v_BatchID]});
        stmt5.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        var stmt6 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne(?)", binds: [v_BatchID]});
        stmt6.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        var stmt7 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise(?)", binds: [v_BatchID]});
        stmt7.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        var stmt8 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_DimDate(?)", binds: [v_BatchID]});
        stmt8.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        var stmt9 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico(?)", binds: [v_BatchID]});
        stmt9.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        var stmt10 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada(?)", binds: [v_BatchID]});
        stmt10.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        var stmt11 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays(?)", binds: [v_BatchID]});
        stmt11.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        var stmt12 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_India(?)", binds: [v_BatchID]});
        stmt12.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // CHECK REQUIRED: Oracle NVL converted to Snowflake COALESCE - validate NULL handling semantics
        // Calculate summary statistics from audit log
        var stmt_success = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_inserted), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'SUCCESS'`,
            binds: [v_BatchID]
        });
        var result_success = stmt_success.execute();
        if (result_success.next()) {
            v_TablesSucceeded = result_success.getColumnValue(1);
            v_TotalRowsInserted = result_success.getColumnValue(2);
        }
        
        var stmt_failed = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_failed), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'FAILED'`,
            binds: [v_BatchID]
        });
        var result_failed = stmt_failed.execute();
        if (result_failed.next()) {
            v_TablesFailed = result_failed.getColumnValue(1);
            v_TotalRowsFailed = result_failed.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // CHECK REQUIRED: Oracle EXTRACT function for timestamp arithmetic converted to Snowflake DATEDIFF - validate calculation behavior
        // Calculate execution time
        var stmt_end = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var result_end = stmt_end.execute();
        result_end.next();
        v_EndTime = result_end.getColumnValue(1);
        
        var stmt_exec_time = snowflake.createStatement({
            sqlText: "SELECT DATEDIFF(SECOND, ?, ?)",
            binds: [v_StartTime, v_EndTime]
        });
        var result_exec_time = stmt_exec_time.execute();
        result_exec_time.next();
        v_ExecutionTime = result_exec_time.getColumnValue(1);
        
        // Log completion
        log_msg += '\n';
        log_msg += '================================================================================\n';
        log_msg += 'bronze Layer ETL Pipeline - Completed Successfully\n';
        log_msg += '================================================================================\n';
        log_msg += 'End Time: ' + v_EndTime + '\n';
        log_msg += 'Execution Time: ' + v_ExecutionTime + ' seconds\n';
        log_msg += 'Tables Processed: ' + v_TablesProcessed + '\n';
        log_msg += 'Tables Succeeded: ' + v_TablesSucceeded + '\n';
        log_msg += 'Tables Failed: ' + v_TablesFailed + '\n';
        log_msg += 'Total Rows Inserted: ' + v_TotalRowsInserted + '\n';
        log_msg += 'Total Rows Failed: ' + (v_TotalRowsFailed || 0) + '\n';
        log_msg += '================================================================================\n';
        
        // Insert master audit record
        var stmt_audit = snowflake.createStatement({
            sqlText: `INSERT INTO bronze.bz_Audit_Log (
                        source_table,
                        target_table,
                        load_timestamp,
                        start_timestamp,
                        end_timestamp,
                        processed_by,
                        processing_time,
                        status,
                        records_processed,
                        records_inserted,
                        records_failed,
                        batch_id,
                        load_type,
                        created_date
                    )
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
            binds: [
                'source_layer.*',
                'bronze.*',
                v_StartTime,
                v_StartTime,
                v_EndTime,
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                v_TotalRowsProcessed,
                v_TotalRowsInserted,
                v_TotalRowsFailed,
                v_BatchID,
                'FULL_REFRESH_ALL_TABLES'
            ]
        });
        stmt_audit.execute();
        
        return log_msg + '\nStatus: ' + v_OverallStatus;
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        var stmt_end_err = snowflake.createStatement({sqlText: "SELECT CURRENT_TIMESTAMP()"});
        var result_end_err = stmt_end_err.execute();
        result_end_err.next();
        v_EndTime = result_end_err.getColumnValue(1);
        
        var stmt_exec_time_err = snowflake.createStatement({
            sqlText: "SELECT DATEDIFF(SECOND, ?, ?)",
            binds: [v_StartTime, v_EndTime]
        });
        var result_exec_time_err = stmt_exec_time_err.execute();
        result_exec_time_err.next();
        v_ExecutionTime = result_exec_time_err.getColumnValue(1);
        
        v_OverallStatus = 'FAILED';
        
        // Log error
        var error_log = '';
        error_log += '\n';
        error_log += '================================================================================\n';
        error_log += 'ERROR in bronze Layer ETL Pipeline\n';
        error_log += '================================================================================\n';
        error_log += 'Error Number: ' + v_ErrorNumber + '\n';
        error_log += 'Error Severity: ' + v_ErrorSeverity + '\n';
        error_log += 'Error State: ' + v_ErrorState + '\n';
        error_log += 'Error Line: ' + v_ErrorLine + '\n';
        error_log += 'Error Message: ' + v_ErrorMessage + '\n';
        error_log += '================================================================================\n';
        
        // Insert error audit record
        var stmt_audit_err = snowflake.createStatement({
            sqlText: `INSERT INTO bronze.bz_Audit_Log (
                        source_table,
                        target_table,
                        load_timestamp,
                        start_timestamp,
                        end_timestamp,
                        processed_by,
                        processing_time,
                        status,
                        error_message,
                        batch_id,
                        load_type,
                        created_date
                    )
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
            binds: [
                'source_layer.*',
                'bronze.*',
                v_StartTime,
                v_StartTime,
                v_EndTime,
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                'Error ' + v_ErrorNumber + ': ' + v_ErrorMessage,
                v_BatchID,
                'FULL_REFRESH_ALL_TABLES'
            ]
        });
        stmt_audit_err.execute();
        
        // Re-throw error
        throw error_log + '\nOriginal Error: ' + v_ErrorMessage;
    }
$$;
