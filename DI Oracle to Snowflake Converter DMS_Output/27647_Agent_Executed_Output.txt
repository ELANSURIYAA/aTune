-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE to Snowflake stored procedure using JavaScript
-- - Translated Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) to Snowflake JavaScript variables
-- - Replaced Oracle SYSTIMESTAMP with Snowflake CURRENT_TIMESTAMP()
-- - Converted Oracle USER function to Snowflake CURRENT_USER()
-- - Replaced Oracle TO_CHAR date formatting with Snowflake TO_VARCHAR and TO_CHAR equivalents
-- - Converted Oracle DBMS_OUTPUT.PUT_LINE to Snowflake JavaScript return statements and logging
-- - Translated Oracle procedure calls to Snowflake CALL statements within JavaScript
-- - Replaced Oracle NVL with Snowflake COALESCE
-- - Converted Oracle EXTRACT function for date arithmetic to Snowflake DATEDIFF
-- - Translated Oracle INSERT INTO statements to Snowflake INSERT with equivalent syntax
-- - Replaced Oracle COMMIT with Snowflake auto-commit behavior
-- - Converted Oracle EXCEPTION WHEN OTHERS to JavaScript try-catch blocks
-- - Replaced Oracle SQLERRM and SQLCODE with JavaScript error object properties
-- - Converted Oracle RAISE_APPLICATION_ERROR to JavaScript throw statement
-- Major Risks / Checks :
-- - Validate date/time function behavior differences between Oracle and Snowflake
-- - Validate NULL handling differences (NVL vs COALESCE behavior)
-- - Validate transaction management (Oracle COMMIT vs Snowflake auto-commit)
-- - Validate error handling and exception propagation
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
EXECUTE AS CALLER
AS
$$
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage = '';
    var v_ErrorNumber = 0;
    var v_ErrorSeverity = 0;
    var v_ErrorState = 0;
    var v_ErrorLine = 0;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser = '';
    var v_BatchID = '';
    
    try {
        // Get current user
        // CHECK REQUIRED:
        // Oracle USER function converted to Snowflake CURRENT_USER().
        // Validate that user context is correctly captured.
        var userStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var userResult = userStmt.execute();
        if (userResult.next()) {
            v_CurrentUser = userResult.getColumnValue(1);
        }
        
        // Generate batch ID
        // CHECK REQUIRED:
        // Oracle TO_CHAR date formatting converted to Snowflake TO_VARCHAR.
        // Validate timestamp format matches expected pattern.
        var batchStmt = snowflake.createStatement({sqlText: "SELECT TO_VARCHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var batchResult = batchStmt.execute();
        if (batchResult.next()) {
            v_BatchID = batchResult.getColumnValue(1);
        }
        
        // Log start of overall process
        var logMessage = '';
        logMessage += '================================================================================\n';
        logMessage += 'bronze Layer ETL Pipeline - Started at: ' + v_BatchID + '\n';
        logMessage += 'Executed by: ' + v_CurrentUser + '\n';
        logMessage += 'Batch ID: ' + v_BatchID + '\n';
        logMessage += '================================================================================\n';
        logMessage += '\n';
        
        // Load Table 1: bz_New_Monthly_HC_Report
        // CHECK REQUIRED:
        // Oracle procedure call syntax converted to Snowflake stored procedure call.
        // Validate parameter mapping and execution behavior.
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask (excludes TIMESTAMP column)
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_SchTask(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_report_392_all(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_DimDate(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_India(?)", binds: [v_BatchID]}).execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Calculate summary statistics from audit log
        // CHECK REQUIRED:
        // Oracle NVL converted to Snowflake COALESCE.
        // Validate NULL handling behavior matches Oracle semantics.
        var successStmt = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_inserted), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'SUCCESS'`,
            binds: [v_BatchID]
        });
        var successResult = successStmt.execute();
        if (successResult.next()) {
            v_TablesSucceeded = successResult.getColumnValue(1);
            v_TotalRowsInserted = successResult.getColumnValue(2);
        }
        
        var failedStmt = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_failed), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'FAILED'`,
            binds: [v_BatchID]
        });
        var failedResult = failedStmt.execute();
        if (failedResult.next()) {
            v_TablesFailed = failedResult.getColumnValue(1);
            v_TotalRowsFailed = failedResult.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // Calculate execution time
        // CHECK REQUIRED:
        // Oracle EXTRACT function converted to Snowflake DATEDIFF.
        // Validate that date/time calculations preserve exact temporal semantics.
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        
        // Log completion
        logMessage += '\n';
        logMessage += '================================================================================\n';
        logMessage += 'bronze Layer ETL Pipeline - Completed Successfully\n';
        logMessage += '================================================================================\n';
        logMessage += 'End Time: ' + v_EndTime.toISOString() + '\n';
        logMessage += 'Execution Time: ' + v_ExecutionTime + ' seconds\n';
        logMessage += 'Tables Processed: ' + v_TablesProcessed + '\n';
        logMessage += 'Tables Succeeded: ' + v_TablesSucceeded + '\n';
        logMessage += 'Tables Failed: ' + v_TablesFailed + '\n';
        logMessage += 'Total Rows Inserted: ' + v_TotalRowsInserted + '\n';
        logMessage += 'Total Rows Failed: ' + (v_TotalRowsFailed || 0) + '\n';
        logMessage += '================================================================================\n';
        
        // Insert master audit record
        var insertStmt = snowflake.createStatement({
            sqlText: `INSERT INTO bronze.bz_Audit_Log (
                        source_table,
                        target_table,
                        load_timestamp,
                        start_timestamp,
                        end_timestamp,
                        processed_by,
                        processing_time,
                        status,
                        records_processed,
                        records_inserted,
                        records_failed,
                        batch_id,
                        load_type,
                        created_date
                    )
                    VALUES (
                        'source_layer.*',
                        'bronze.*',
                        TO_TIMESTAMP_NTZ(?),
                        TO_TIMESTAMP_NTZ(?),
                        CURRENT_TIMESTAMP(),
                        ?,
                        ?,
                        ?,
                        ?,
                        ?,
                        ?,
                        ?,
                        'FULL_REFRESH_ALL_TABLES',
                        CURRENT_TIMESTAMP()
                    )`,
            binds: [
                v_BatchID,
                v_BatchID,
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                v_TotalRowsProcessed,
                v_TotalRowsInserted,
                v_TotalRowsFailed,
                v_BatchID
            ]
        });
        insertStmt.execute();
        
        return logMessage + 'SUCCESS: bronze Layer ETL Pipeline completed successfully.';
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        v_OverallStatus = 'FAILED';
        
        // Log error
        var errorLog = '';
        errorLog += '\n';
        errorLog += '================================================================================\n';
        errorLog += 'ERROR in bronze Layer ETL Pipeline\n';
        errorLog += '================================================================================\n';
        errorLog += 'Error Number: ' + v_ErrorNumber + '\n';
        errorLog += 'Error Severity: ' + v_ErrorSeverity + '\n';
        errorLog += 'Error State: ' + v_ErrorState + '\n';
        errorLog += 'Error Line: ' + v_ErrorLine + '\n';
        errorLog += 'Error Message: ' + v_ErrorMessage + '\n';
        errorLog += '================================================================================\n';
        
        // Insert error audit record
        try {
            var errorInsertStmt = snowflake.createStatement({
                sqlText: `INSERT INTO bronze.bz_Audit_Log (
                            source_table,
                            target_table,
                            load_timestamp,
                            start_timestamp,
                            end_timestamp,
                            processed_by,
                            processing_time,
                            status,
                            error_message,
                            batch_id,
                            load_type,
                            created_date
                        )
                        VALUES (
                            'source_layer.*',
                            'bronze.*',
                            TO_TIMESTAMP_NTZ(?),
                            TO_TIMESTAMP_NTZ(?),
                            CURRENT_TIMESTAMP(),
                            ?,
                            ?,
                            ?,
                            ?,
                            ?,
                            'FULL_REFRESH_ALL_TABLES',
                            CURRENT_TIMESTAMP()
                        )`,
                binds: [
                    v_BatchID,
                    v_BatchID,
                    v_CurrentUser,
                    v_ExecutionTime,
                    v_OverallStatus,
                    'Error ' + v_ErrorNumber + ': ' + v_ErrorMessage,
                    v_BatchID
                ]
            });
            errorInsertStmt.execute();
        } catch (insertErr) {
            errorLog += 'Failed to insert error audit record: ' + insertErr.message + '\n';
        }
        
        // Re-throw error
        throw new Error(errorLog + v_ErrorMessage);
    }
$$;
