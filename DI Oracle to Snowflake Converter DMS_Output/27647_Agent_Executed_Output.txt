-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE into Snowflake CREATE OR REPLACE PROCEDURE with JavaScript
-- - Replaced Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) with Snowflake JavaScript equivalents
-- - Converted Oracle SYSTIMESTAMP to Snowflake CURRENT_TIMESTAMP()
-- - Replaced Oracle USER function with CURRENT_USER()
-- - Converted Oracle TO_CHAR date formatting to Snowflake TO_VARCHAR with equivalent format strings
-- - Replaced Oracle DBMS_OUTPUT.PUT_LINE with Snowflake JavaScript return statements and logging
-- - Converted Oracle procedure calls into Snowflake CALL statements within JavaScript
-- - Translated Oracle SELECT INTO statements into Snowflake JavaScript result set processing
-- - Replaced Oracle NVL function with Snowflake COALESCE
-- - Converted Oracle EXTRACT function for timestamp arithmetic to Snowflake DATEDIFF
-- - Translated Oracle INSERT INTO statements into Snowflake INSERT with JavaScript variable binding
-- - Replaced Oracle COMMIT with Snowflake auto-commit behavior
-- - Converted Oracle EXCEPTION WHEN OTHERS block into JavaScript try-catch error handling
-- - Replaced Oracle SQLERRM and SQLCODE with JavaScript error object properties
-- - Converted Oracle RAISE_APPLICATION_ERROR into Snowflake JavaScript throw statement
-- Major Risks / Checks :
-- - Validate timestamp arithmetic conversion from Oracle EXTRACT to Snowflake DATEDIFF
-- - Validate NULL handling differences between Oracle NVL and Snowflake COALESCE
-- - Validate error handling behavior between Oracle exceptions and JavaScript try-catch
-- - Validate transaction commit behavior differences
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
AS
$$
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage;
    var v_ErrorNumber;
    var v_ErrorSeverity;
    var v_ErrorState;
    var v_ErrorLine;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser;
    var v_BatchID;
    var logOutput = '';
    
    try {
        // Get current user
        var userStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var userResult = userStmt.execute();
        userResult.next();
        v_CurrentUser = userResult.getColumnValue(1);
        
        // Get current timestamp and format batch ID
        var timestampStmt = snowflake.createStatement({sqlText: "SELECT TO_VARCHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var timestampResult = timestampStmt.execute();
        timestampResult.next();
        v_BatchID = timestampResult.getColumnValue(1);
        
        // Log start of overall process
        logOutput += '================================================================================\n';
        logOutput += 'bronze Layer ETL Pipeline - Started at: ' + v_BatchID + '\n';
        logOutput += 'Executed by: ' + v_CurrentUser + '\n';
        logOutput += 'Batch ID: ' + v_BatchID + '\n';
        logOutput += '================================================================================\n';
        logOutput += '\n';
        
        // Load Table 1: bz_New_Monthly_HC_Report
        var stmt1 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report(?)", binds: [v_BatchID]});
        stmt1.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask (excludes TIMESTAMP column)
        var stmt2 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_SchTask(?)", binds: [v_BatchID]});
        stmt2.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        var stmt3 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info(?)", binds: [v_BatchID]});
        stmt3.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        var stmt4 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New(?)", binds: [v_BatchID]});
        stmt4.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        var stmt5 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_report_392_all(?)", binds: [v_BatchID]});
        stmt5.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        var stmt6 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne(?)", binds: [v_BatchID]});
        stmt6.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        var stmt7 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise(?)", binds: [v_BatchID]});
        stmt7.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        var stmt8 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_DimDate(?)", binds: [v_BatchID]});
        stmt8.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        var stmt9 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico(?)", binds: [v_BatchID]});
        stmt9.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        var stmt10 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada(?)", binds: [v_BatchID]});
        stmt10.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        var stmt11 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays(?)", binds: [v_BatchID]});
        stmt11.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        var stmt12 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_India(?)", binds: [v_BatchID]});
        stmt12.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Calculate summary statistics from audit log
        var successStmt = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_inserted), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'SUCCESS'`,
            binds: [v_BatchID]
        });
        var successResult = successStmt.execute();
        if (successResult.next()) {
            v_TablesSucceeded = successResult.getColumnValue(1);
            v_TotalRowsInserted = successResult.getColumnValue(2);
        }
        
        var failedStmt = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_failed), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'FAILED'`,
            binds: [v_BatchID]
        });
        var failedResult = failedStmt.execute();
        if (failedResult.next()) {
            v_TablesFailed = failedResult.getColumnValue(1);
            v_TotalRowsFailed = failedResult.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // Calculate execution time
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        
        // Get formatted end time
        var endTimeStmt = snowflake.createStatement({sqlText: "SELECT TO_VARCHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var endTimeResult = endTimeStmt.execute();
        endTimeResult.next();
        var v_EndTimeFormatted = endTimeResult.getColumnValue(1);
        
        // Log completion
        logOutput += '\n';
        logOutput += '================================================================================\n';
        logOutput += 'bronze Layer ETL Pipeline - Completed Successfully\n';
        logOutput += '================================================================================\n';
        logOutput += 'End Time: ' + v_EndTimeFormatted + '\n';
        logOutput += 'Execution Time: ' + v_ExecutionTime + ' seconds\n';
        logOutput += 'Tables Processed: ' + v_TablesProcessed + '\n';
        logOutput += 'Tables Succeeded: ' + v_TablesSucceeded + '\n';
        logOutput += 'Tables Failed: ' + v_TablesFailed + '\n';
        logOutput += 'Total Rows Inserted: ' + v_TotalRowsInserted + '\n';
        logOutput += 'Total Rows Failed: ' + (v_TotalRowsFailed || 0) + '\n';
        logOutput += '================================================================================\n';
        
        // Insert master audit record
        var auditStmt = snowflake.createStatement({
            sqlText: `INSERT INTO bronze.bz_Audit_Log (
                        source_table,
                        target_table,
                        load_timestamp,
                        start_timestamp,
                        end_timestamp,
                        processed_by,
                        processing_time,
                        status,
                        records_processed,
                        records_inserted,
                        records_failed,
                        batch_id,
                        load_type,
                        created_date
                    )
                    VALUES (?, ?, ?, ?, CURRENT_TIMESTAMP(), ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
            binds: [
                'source_layer.*',
                'bronze.*',
                v_BatchID,
                v_BatchID,
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                v_TotalRowsProcessed,
                v_TotalRowsInserted,
                v_TotalRowsFailed,
                v_BatchID,
                'FULL_REFRESH_ALL_TABLES'
            ]
        });
        auditStmt.execute();
        
        return logOutput;
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        v_OverallStatus = 'FAILED';
        
        // Log error
        logOutput += '\n';
        logOutput += '================================================================================\n';
        logOutput += 'ERROR in bronze Layer ETL Pipeline\n';
        logOutput += '================================================================================\n';
        logOutput += 'Error Number: ' + v_ErrorNumber + '\n';
        logOutput += 'Error Severity: ' + v_ErrorSeverity + '\n';
        logOutput += 'Error State: ' + v_ErrorState + '\n';
        logOutput += 'Error Line: ' + v_ErrorLine + '\n';
        logOutput += 'Error Message: ' + v_ErrorMessage + '\n';
        logOutput += '================================================================================\n';
        
        // Insert error audit record
        try {
            var errorAuditStmt = snowflake.createStatement({
                sqlText: `INSERT INTO bronze.bz_Audit_Log (
                            source_table,
                            target_table,
                            load_timestamp,
                            start_timestamp,
                            end_timestamp,
                            processed_by,
                            processing_time,
                            status,
                            error_message,
                            batch_id,
                            load_type,
                            created_date
                        )
                        VALUES (?, ?, ?, ?, CURRENT_TIMESTAMP(), ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
                binds: [
                    'source_layer.*',
                    'bronze.*',
                    v_BatchID,
                    v_BatchID,
                    v_CurrentUser,
                    v_ExecutionTime,
                    v_OverallStatus,
                    'Error ' + v_ErrorNumber + ': ' + v_ErrorMessage,
                    v_BatchID,
                    'FULL_REFRESH_ALL_TABLES'
                ]
            });
            errorAuditStmt.execute();
        } catch (auditErr) {
            logOutput += 'Failed to insert error audit record: ' + auditErr.message + '\n';
        }
        
        // Re-throw error
        throw err;
    }
$$;
