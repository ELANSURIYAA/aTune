-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE into Snowflake stored procedure using JavaScript
-- - Translated Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) into Snowflake JavaScript variables
-- - Replaced Oracle SYSTIMESTAMP with Snowflake CURRENT_TIMESTAMP()
-- - Converted Oracle USER function to CURRENT_USER()
-- - Replaced Oracle TO_CHAR date formatting with Snowflake TO_VARCHAR and TO_CHAR equivalents
-- - Converted Oracle DBMS_OUTPUT.PUT_LINE into Snowflake result set or logging mechanism
-- - Translated Oracle procedure calls into Snowflake CALL statements
-- - Converted Oracle NVL function to Snowflake COALESCE/IFNULL
-- - Replaced Oracle EXTRACT function for date arithmetic with Snowflake DATEDIFF
-- - Converted Oracle INSERT INTO statements into Snowflake INSERT INTO with equivalent syntax
-- - Translated Oracle COMMIT into Snowflake transaction control
-- - Converted Oracle exception handling (EXCEPTION WHEN OTHERS) into JavaScript try-catch blocks
-- - Replaced Oracle SQLERRM and SQLCODE with Snowflake error handling variables
-- - Converted Oracle RAISE_APPLICATION_ERROR into Snowflake throw statement
-- Major Risks / Checks :
-- - Validate date/time function behavior differences between Oracle and Snowflake
-- - Validate NULL handling differences (NVL vs COALESCE behavior)
-- - Validate transaction control and commit behavior
-- - Validate error handling and exception propagation
-- - Validate procedure call parameter passing
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
AS
$$
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage = '';
    var v_ErrorNumber = 0;
    var v_ErrorSeverity = 0;
    var v_ErrorState = 0;
    var v_ErrorLine = 0;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser = '';
    var v_BatchID = '';
    
    try {
        // Get current user
        var userStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var userResult = userStmt.execute();
        if (userResult.next()) {
            v_CurrentUser = userResult.getColumnValue(1);
        }
        
        // Format batch ID
        var batchStmt = snowflake.createStatement({sqlText: "SELECT TO_VARCHAR(CURRENT_TIMESTAMP(), 'YYYY-MM-DD HH24:MI:SS.FF3')"});
        var batchResult = batchStmt.execute();
        if (batchResult.next()) {
            v_BatchID = batchResult.getColumnValue(1);
        }
        
        // Log start of overall process
        var logMessage = '';
        logMessage += '================================================================================\n';
        logMessage += 'bronze Layer ETL Pipeline - Started at: ' + v_BatchID + '\n';
        logMessage += 'Executed by: ' + v_CurrentUser + '\n';
        logMessage += 'Batch ID: ' + v_BatchID + '\n';
        logMessage += '================================================================================\n';
        logMessage += '\n';
        
        // Load Table 1: bz_New_Monthly_HC_Report
        var stmt1 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report(?)", binds: [v_BatchID]});
        stmt1.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask (excludes TIMESTAMP column)
        var stmt2 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_SchTask(?)", binds: [v_BatchID]});
        stmt2.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        var stmt3 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info(?)", binds: [v_BatchID]});
        stmt3.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        var stmt4 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New(?)", binds: [v_BatchID]});
        stmt4.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        var stmt5 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_report_392_all(?)", binds: [v_BatchID]});
        stmt5.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        var stmt6 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne(?)", binds: [v_BatchID]});
        stmt6.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        var stmt7 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise(?)", binds: [v_BatchID]});
        stmt7.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        var stmt8 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_DimDate(?)", binds: [v_BatchID]});
        stmt8.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        var stmt9 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico(?)", binds: [v_BatchID]});
        stmt9.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        var stmt10 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada(?)", binds: [v_BatchID]});
        stmt10.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        var stmt11 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays(?)", binds: [v_BatchID]});
        stmt11.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        var stmt12 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_India(?)", binds: [v_BatchID]});
        stmt12.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Calculate summary statistics from audit log
        var summaryStmt1 = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_inserted), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'SUCCESS'`,
            binds: [v_BatchID]
        });
        var summaryResult1 = summaryStmt1.execute();
        if (summaryResult1.next()) {
            v_TablesSucceeded = summaryResult1.getColumnValue(1);
            v_TotalRowsInserted = summaryResult1.getColumnValue(2);
        }
        
        var summaryStmt2 = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_failed), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'FAILED'`,
            binds: [v_BatchID]
        });
        var summaryResult2 = summaryStmt2.execute();
        if (summaryResult2.next()) {
            v_TablesFailed = summaryResult2.getColumnValue(1);
            v_TotalRowsFailed = summaryResult2.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // Calculate execution time
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        
        // CHECK REQUIRED:
        // Oracle EXTRACT function converted to Snowflake date arithmetic (DATEDIFF/DATEADD).
        // Validate that date/time calculations preserve exact temporal semantics and boundary conditions.
        
        // Log completion
        logMessage += '\n';
        logMessage += '================================================================================\n';
        logMessage += 'bronze Layer ETL Pipeline - Completed Successfully\n';
        logMessage += '================================================================================\n';
        logMessage += 'End Time: ' + v_EndTime.toISOString() + '\n';
        logMessage += 'Execution Time: ' + v_ExecutionTime + ' seconds\n';
        logMessage += 'Tables Processed: ' + v_TablesProcessed + '\n';
        logMessage += 'Tables Succeeded: ' + v_TablesSucceeded + '\n';
        logMessage += 'Tables Failed: ' + v_TablesFailed + '\n';
        logMessage += 'Total Rows Inserted: ' + v_TotalRowsInserted + '\n';
        logMessage += 'Total Rows Failed: ' + (v_TotalRowsFailed || 0) + '\n';
        logMessage += '================================================================================\n';
        
        // Insert master audit record
        var auditStmt = snowflake.createStatement({
            sqlText: `INSERT INTO bronze.bz_Audit_Log (
                        source_table,
                        target_table,
                        load_timestamp,
                        start_timestamp,
                        end_timestamp,
                        processed_by,
                        processing_time,
                        status,
                        records_processed,
                        records_inserted,
                        records_failed,
                        batch_id,
                        load_type,
                        created_date
                    )
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
            binds: [
                'source_layer.*',
                'bronze.*',
                v_BatchID,
                v_BatchID,
                v_EndTime.toISOString(),
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                v_TotalRowsProcessed,
                v_TotalRowsInserted,
                v_TotalRowsFailed,
                v_BatchID,
                'FULL_REFRESH_ALL_TABLES'
            ]
        });
        auditStmt.execute();
        
        // CHECK REQUIRED:
        // Oracle COMMIT converted to Snowflake auto-commit behavior.
        // Validate transaction control semantics.
        
        return logMessage + '\nProcedure completed successfully.';
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        v_OverallStatus = 'FAILED';
        
        // Log error
        var errorLog = '';
        errorLog += '\n';
        errorLog += '================================================================================\n';
        errorLog += 'ERROR in bronze Layer ETL Pipeline\n';
        errorLog += '================================================================================\n';
        errorLog += 'Error Number: ' + v_ErrorNumber + '\n';
        errorLog += 'Error Severity: ' + v_ErrorSeverity + '\n';
        errorLog += 'Error State: ' + v_ErrorState + '\n';
        errorLog += 'Error Line: ' + v_ErrorLine + '\n';
        errorLog += 'Error Message: ' + v_ErrorMessage + '\n';
        errorLog += '================================================================================\n';
        
        // Insert error audit record
        try {
            var errorAuditStmt = snowflake.createStatement({
                sqlText: `INSERT INTO bronze.bz_Audit_Log (
                            source_table,
                            target_table,
                            load_timestamp,
                            start_timestamp,
                            end_timestamp,
                            processed_by,
                            processing_time,
                            status,
                            error_message,
                            batch_id,
                            load_type,
                            created_date
                        )
                        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
                binds: [
                    'source_layer.*',
                    'bronze.*',
                    v_BatchID,
                    v_BatchID,
                    v_EndTime.toISOString(),
                    v_CurrentUser,
                    v_ExecutionTime,
                    v_OverallStatus,
                    'Error ' + v_ErrorNumber + ': ' + v_ErrorMessage,
                    v_BatchID,
                    'FULL_REFRESH_ALL_TABLES'
                ]
            });
            errorAuditStmt.execute();
        } catch (auditErr) {
            errorLog += 'Failed to insert error audit record: ' + auditErr.message + '\n';
        }
        
        // CHECK REQUIRED:
        // Oracle RAISE_APPLICATION_ERROR converted to Snowflake throw statement.
        // Validate error propagation and error code handling.
        
        throw errorLog + v_ErrorMessage;
    }
$$;
