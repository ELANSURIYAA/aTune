-- =========================================================
-- CONVERSION LOG
-- =========================================================
-- Input Type : Oracle Stored Procedure
-- Target Platform : Snowflake
-- Conversion Approach :
-- - Converted Oracle CREATE OR REPLACE PROCEDURE to Snowflake stored procedure using JavaScript
-- - Replaced Oracle variable declarations (VARCHAR2, NUMBER, TIMESTAMP) with Snowflake JavaScript equivalents
-- - Converted SYSTIMESTAMP to CURRENT_TIMESTAMP()
-- - Replaced USER with CURRENT_USER()
-- - Converted TO_CHAR date formatting to Snowflake TO_VARCHAR with appropriate format strings
-- - Replaced DBMS_OUTPUT.PUT_LINE with JavaScript return statements and logging approach
-- - Converted Oracle procedure calls to Snowflake CALL statements using snowflake.execute()
-- - Replaced NVL with COALESCE in SQL queries
-- - Converted EXTRACT functions for date arithmetic to Snowflake DATEDIFF
-- - Replaced Oracle INSERT INTO with Snowflake INSERT INTO using snowflake.execute()
-- - Converted COMMIT to Snowflake transaction handling (auto-commit behavior)
-- - Replaced SQLERRM and SQLCODE with JavaScript try-catch error handling
-- - Converted RAISE_APPLICATION_ERROR to JavaScript throw statement
-- - Transformed Oracle exception handling block to JavaScript try-catch-finally
-- Major Risks / Checks :
-- - Validate that child procedures exist in Snowflake (bronze.usp_Load_bz_* procedures)
-- - Validate audit table bronze.bz_Audit_Log exists with matching schema
-- - Validate date/time arithmetic conversion accuracy
-- - Validate transaction behavior differences between Oracle COMMIT and Snowflake auto-commit
-- - Validate error handling and logging consistency
-- =========================================================

CREATE OR REPLACE PROCEDURE bronze.usp_Load_bronze_Layer_Full()
RETURNS VARCHAR
LANGUAGE JAVASCRIPT
EXECUTE AS CALLER
AS
$$
    // Variable declarations
    var v_ProcedureName = 'bronze.usp_Load_bronze_Layer_Full';
    var v_StartTime = new Date();
    var v_EndTime;
    var v_ExecutionTime;
    var v_OverallStatus = 'SUCCESS';
    var v_ErrorMessage = '';
    var v_ErrorNumber = 0;
    var v_ErrorSeverity = 0;
    var v_ErrorState = 0;
    var v_ErrorLine = 0;
    var v_TotalRowsProcessed = 0;
    var v_TotalRowsInserted = 0;
    var v_TotalRowsFailed = 0;
    var v_TablesProcessed = 0;
    var v_TablesSucceeded = 0;
    var v_TablesFailed = 0;
    var v_CurrentUser = '';
    var v_BatchID = '';
    var logMessages = [];
    
    try {
        // Get current user
        var userStmt = snowflake.createStatement({sqlText: "SELECT CURRENT_USER()"});
        var userResult = userStmt.execute();
        if (userResult.next()) {
            v_CurrentUser = userResult.getColumnValue(1);
        }
        
        // Generate batch ID
        v_BatchID = v_StartTime.toISOString().replace('T', ' ').substring(0, 23);
        
        // Log start of overall process
        logMessages.push('================================================================================');
        logMessages.push('bronze Layer ETL Pipeline - Started at: ' + v_BatchID);
        logMessages.push('Executed by: ' + v_CurrentUser);
        logMessages.push('Batch ID: ' + v_BatchID);
        logMessages.push('================================================================================');
        logMessages.push('');
        
        // Load Table 1: bz_New_Monthly_HC_Report
        var stmt1 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_New_Monthly_HC_Report(?)", binds: [v_BatchID]});
        stmt1.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 2: bz_SchTask (excludes TIMESTAMP column)
        var stmt2 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_SchTask(?)", binds: [v_BatchID]});
        stmt2.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 3: bz_Hiring_Initiator_Project_Info
        var stmt3 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Hiring_Initiator_Project_Info(?)", binds: [v_BatchID]});
        stmt3.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 4: bz_Timesheet_New
        var stmt4 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_Timesheet_New(?)", binds: [v_BatchID]});
        stmt4.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 5: bz_report_392_all
        var stmt5 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_report_392_all(?)", binds: [v_BatchID]});
        stmt5.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 6: bz_vw_billing_timesheet_daywise_ne
        var stmt6 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_billing_timesheet_daywise_ne(?)", binds: [v_BatchID]});
        stmt6.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 7: bz_vw_consultant_timesheet_daywise
        var stmt7 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_vw_consultant_timesheet_daywise(?)", binds: [v_BatchID]});
        stmt7.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 8: bz_DimDate
        var stmt8 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_DimDate(?)", binds: [v_BatchID]});
        stmt8.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 9: bz_holidays_Mexico
        var stmt9 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Mexico(?)", binds: [v_BatchID]});
        stmt9.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 10: bz_holidays_Canada
        var stmt10 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_Canada(?)", binds: [v_BatchID]});
        stmt10.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 11: bz_holidays
        var stmt11 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays(?)", binds: [v_BatchID]});
        stmt11.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Load Table 12: bz_holidays_India
        var stmt12 = snowflake.createStatement({sqlText: "CALL bronze.usp_Load_bz_holidays_India(?)", binds: [v_BatchID]});
        stmt12.execute();
        v_TablesProcessed = v_TablesProcessed + 1;
        
        // Calculate summary statistics from audit log
        var successStmt = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_inserted), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'SUCCESS'`,
            binds: [v_BatchID]
        });
        var successResult = successStmt.execute();
        if (successResult.next()) {
            v_TablesSucceeded = successResult.getColumnValue(1);
            v_TotalRowsInserted = successResult.getColumnValue(2);
        }
        
        var failedStmt = snowflake.createStatement({
            sqlText: `SELECT 
                        COUNT(*),
                        COALESCE(SUM(records_failed), 0)
                      FROM bronze.bz_Audit_Log
                      WHERE batch_id = ?
                        AND status = 'FAILED'`,
            binds: [v_BatchID]
        });
        var failedResult = failedStmt.execute();
        if (failedResult.next()) {
            v_TablesFailed = failedResult.getColumnValue(1);
            v_TotalRowsFailed = failedResult.getColumnValue(2);
        }
        
        v_TotalRowsProcessed = v_TotalRowsInserted + (v_TotalRowsFailed || 0);
        
        // Calculate execution time
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        
        // Log completion
        logMessages.push('');
        logMessages.push('================================================================================');
        logMessages.push('bronze Layer ETL Pipeline - Completed Successfully');
        logMessages.push('================================================================================');
        logMessages.push('End Time: ' + v_EndTime.toISOString().replace('T', ' ').substring(0, 23));
        logMessages.push('Execution Time: ' + v_ExecutionTime.toString() + ' seconds');
        logMessages.push('Tables Processed: ' + v_TablesProcessed.toString());
        logMessages.push('Tables Succeeded: ' + v_TablesSucceeded.toString());
        logMessages.push('Tables Failed: ' + v_TablesFailed.toString());
        logMessages.push('Total Rows Inserted: ' + v_TotalRowsInserted.toString());
        logMessages.push('Total Rows Failed: ' + (v_TotalRowsFailed || 0).toString());
        logMessages.push('================================================================================');
        
        // Insert master audit record
        var auditStmt = snowflake.createStatement({
            sqlText: `INSERT INTO bronze.bz_Audit_Log (
                        source_table,
                        target_table,
                        load_timestamp,
                        start_timestamp,
                        end_timestamp,
                        processed_by,
                        processing_time,
                        status,
                        records_processed,
                        records_inserted,
                        records_failed,
                        batch_id,
                        load_type,
                        created_date
                    )
                    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
            binds: [
                'source_layer.*',
                'bronze.*',
                v_StartTime.toISOString(),
                v_StartTime.toISOString(),
                v_EndTime.toISOString(),
                v_CurrentUser,
                v_ExecutionTime,
                v_OverallStatus,
                v_TotalRowsProcessed,
                v_TotalRowsInserted,
                v_TotalRowsFailed,
                v_BatchID,
                'FULL_REFRESH_ALL_TABLES'
            ]
        });
        auditStmt.execute();
        
        return logMessages.join('\n');
        
    } catch (err) {
        // Capture error details
        v_ErrorMessage = err.message;
        v_ErrorNumber = err.code || 0;
        v_ErrorSeverity = 0;
        v_ErrorState = 0;
        v_ErrorLine = err.lineNumber || 0;
        
        v_EndTime = new Date();
        v_ExecutionTime = (v_EndTime - v_StartTime) / 1000;
        v_OverallStatus = 'FAILED';
        
        // Log error
        logMessages.push('');
        logMessages.push('================================================================================');
        logMessages.push('ERROR in bronze Layer ETL Pipeline');
        logMessages.push('================================================================================');
        logMessages.push('Error Number: ' + v_ErrorNumber.toString());
        logMessages.push('Error Severity: ' + v_ErrorSeverity.toString());
        logMessages.push('Error State: ' + v_ErrorState.toString());
        logMessages.push('Error Line: ' + v_ErrorLine.toString());
        logMessages.push('Error Message: ' + v_ErrorMessage);
        logMessages.push('================================================================================');
        
        // Insert error audit record
        try {
            var errorAuditStmt = snowflake.createStatement({
                sqlText: `INSERT INTO bronze.bz_Audit_Log (
                            source_table,
                            target_table,
                            load_timestamp,
                            start_timestamp,
                            end_timestamp,
                            processed_by,
                            processing_time,
                            status,
                            error_message,
                            batch_id,
                            load_type,
                            created_date
                        )
                        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP())`,
                binds: [
                    'source_layer.*',
                    'bronze.*',
                    v_StartTime.toISOString(),
                    v_StartTime.toISOString(),
                    v_EndTime.toISOString(),
                    v_CurrentUser,
                    v_ExecutionTime,
                    v_OverallStatus,
                    'Error ' + v_ErrorNumber.toString() + ': ' + v_ErrorMessage,
                    v_BatchID,
                    'FULL_REFRESH_ALL_TABLES'
                ]
            });
            errorAuditStmt.execute();
        } catch (auditErr) {
            logMessages.push('Failed to insert error audit record: ' + auditErr.message);
        }
        
        // Re-throw error
        throw new Error(v_ErrorMessage);
    }
$$;
